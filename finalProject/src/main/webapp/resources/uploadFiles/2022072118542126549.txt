DROP TABLE BOARD;
DROP TABLE BOARD_REPLY;
DROP TABLE BOARD_LIKE;
DROP TABLE BOARD REPORT;
DROP TABLE REPLY_REPORT;

DROP SEQUENCE



--게시판 테이블
CREATE TABLE BOARD(
    NO NUMBER PRIMARY KEY,
    USER_NO NUMBER NOT NULL,
    CATEGORY VARCHAR2(100) NOT NULL,
    TITLE VARCHAR2(4000) NOT NULL,
    CONTENT VARCHAR2(4000) NOT NULL,
    COUNT NUMBER NOT NULL,
    LIKE_COUNT NUMBER NOT NULL,
    CHANGE_NAME VARCHAR2(4000),
    ORIGIN_NAME VARCHAR2(4000),
    "DATE" DATE DEFAULT SYSDATE,
    STATUS VARCHAR2(1) DEFAULT 'Y'
    );

--게시판 댓글 테이블
CREATE TABLE BOARD_REPLY(
    NO NUMBER PRIMARY KEY,
    USER_NO NUMBER NOT NULL,
    CONTENT VARCHAR2(4000) NOT NULL,
    BOARD_NO NUMBER NOT NULL,
    "DATE" DATE DEFAULT SYSDATE,
    STATUS VARCHAR2(1) DEFAULT 'Y' NOT NULL,
    CHANGE_NAME VARCHAR2(4000),
    ORIGIN_NAME VARCHAR2(4000)
    );
    

CREATE TABLE BOARD_LIKE(
    NO NUMBER PRIMARY KEY,
    USER_NO NUMBER NOT NULL,
    BOARD_NO NUMBER NOT NULL,
    STATUS VARCHAR2(1) DEFAULT 'Y' NOT NULL
    );
    
CREATE TABLE BOARD_REPORT(
   NO NUMBER PRIMARY KEY,
   USER_NO NUMBER NOT NULL,
   "DATE" DATE DEFAULT SYSDATE NOT NULL,
   REASON VARCHAR2(4000) NOT NULL,
   BOARD_NO NUMBER NOT NULL
   );
   
CREATE TABLE REPLY_REPORT(
    NO NUMBER PRIMARY KEY,
    USER_NO NUMBER NOT NULL,
    "DATE" DATE DEFAULT SYSDATE NOT NULL,
    REASON VARCHAR2(4000) NOT NULL,
    REPLY_NO NUMBER NOT NULL
    );
